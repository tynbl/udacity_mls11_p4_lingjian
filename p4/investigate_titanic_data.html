def age_to_group(age):
    if age >= 70:
        return '70+'
    elif age >= 60:
        return '60+'
    elif age >= 50:
        return '50+'
    elif age >= 40:
        return '40+'
    elif age >= 30:
        return '30+'
    elif age >= 20:
        return '20+'
    elif age >= 10:
        return '10+'
    elif age >= 0:
        return '0+'
    else:
        return ''

# 
def age_to_group_fix(age_name_df):
    if age_name_df['Age'] >= 70:
        return '70+'
    elif age_name_df['Age'] >= 60:
        return '60+'
    elif age_name_df['Age'] >= 50:
        return '50+'
    elif age_name_df['Age'] >= 40:
        return '40+'
    elif age_name_df['Age'] >= 30:
        return '30+'
    elif age_name_df['Age'] >= 20:
        return '20+'
    elif age_name_df['Age'] >= 10:
        return '10+'
    elif age_name_df['Age'] >= 0:
        return '0+'
    elif 'Master'in age_name_df['Name']: # Master  4.574167
        return '0+'
    elif 'Miss'in age_name_df['Name']: # Miss  21.773973
        return '20+'
    else:
        return ''

# 
def sex_to_chn(sex):
    if sex == 'male':
        return ''
    elif sex == 'female':
        return ''
    else:
        return ''

# 
def sex_to_code(sex):
    if sex == 'male':
        return 0
    elif sex == 'female':
        return 1
    else:
        return -1

# 
def pclass_to_chn(pclass):
    if pclass == 1:
        return ''
    elif pclass == 2:
        return ''
    elif pclass == 3:
        return ''
    else:
        return ''

# 
def pclass_to_code(pclass):
    if pclass == 1:
        return 2
    elif pclass == 2:
        return 1
    elif pclass == 3:
        return 0
    else:
        return -1

# 
def survived_to_chn(survived):
    if survived == 0:
        return ''
    elif survived == 1:
        return ''
    else:
        return ''

def groups_survived_rate_charts(groupby_c1=None, groupby_c2=None):
    '''
         
        (DataFrame)
    '''
    #  x 
    var_rot = 'horizontal'

    # 
    if groupby_c1 == None and groupby_c2 == None:
        return print('')
    elif groupby_c1 != None and groupby_c2 == None:
        groupby_cs = [groupby_c1, '']
    elif groupby_c1 == None and groupby_c2 != None:
        groupby_cs = [groupby_c2, '']
    else:
        groupby_cs = [groupby_c1, groupby_c2, '']
        var_rot = 'vertical'

    # 
    groups_data = passengers_df.groupby(groupby_cs).size()
    # () 
    groups_data_df = groups_data.unstack()
    #  
    groups_data_df.insert(2, '', groups_data_df[''] / groups_data_df.sum(axis=1))
    groups_data_df.columns.name = None  # 
    
    # 
    groups_data_df[['', '']].plot(kind='bar', stacked=True, alpha=0.6)
    plt.ylabel("")
    # 
    groups_data_df[''].plot(secondary_y=True, kind='line', style='gd-', rot=var_rot)
    plt.ylabel("")

    plt.show()

    # 
    return groups_data_df
